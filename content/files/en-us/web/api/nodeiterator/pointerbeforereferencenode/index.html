The NodeIterator.pointerBeforeReferenceNode read-only property returns a {{domxref("Boolean")}} flag that indicates whether the {{domxref("NodeFilter")}} is anchored before (if this value is true) or after (if this value is false) the anchor node indicated by the {{domxref("NodeIterator.referenceNode")}} property.

Syntax

flag = nodeIterator.pointerBeforeReferenceNode;

Example

var nodeIterator = document.createNodeIterator(
    document.body,
    NodeFilter.SHOW_ELEMENT,
    { acceptNode: function(node) { return NodeFilter.FILTER_ACCEPT; } },
    false
);
flag = nodeIterator.pointerBeforeReferenceNode;
